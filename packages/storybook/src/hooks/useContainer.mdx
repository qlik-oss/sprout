# useContainer

The `useContainer` hook will help you to track the width of a container element.
It's an integration of `@qlik/sprout-css-modules` sprout.container for react.

## API

```tsx
//type BoxSizing = "content-box" | "border-box" | "device-pixel-content-box";
import { useContainer } from "@qlik/sprout-css-hooks";

function MyComponent({id?: string, ...props}) {
  const container = useContainer({ boxSizing: "border-box" });

 return (
  <div ref={container.ref} className={container.classNames({
    s: ["padding_4"],
    m: ["padding_8"],
    l: ["padding_16"],
  })}>
    <Button justified={container.get<boolean>({ xxs: true, m: false })} />;
  </div>
 )
};
```
